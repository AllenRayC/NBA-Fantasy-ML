
USE NBAfantasyML

CREATE TABLE Yearly_Player_Stats
    (UNIQUE_ID AUTO_INCREMENT,
    personId VARCHAR(40),
    seasonYear VARCHAR(40),
    FG FLOAT,
    FGA FLOAT,
    FGP FLOAT,
    TPP FLOAT,
    FT FLOAT,
    FTP FLOAT,
    TRB FLOAT,
    AST FLOAT,
    STL FLOAT,
    BLK FLOAT,
    TOV FLOAT,
    PTG FLOAT,
    PRIMARY KEY (personId))

CREATE TABLE Yearly_Player_Stats
    (UNIQUE_ID VARCHAR(40),
    personId VARCHAR(40),
    seasonYear VARCHAR(40),
    FG FLOAT,
    FGA FLOAT,
    FG% FLOAT,
    3P FLOAT,
    FT FLOAT,
    FT% FLOAT,
    TRB FLOAT,
    AST FLOAT,
    STL FLOAT,
    BLK FLOAT,
    TOV FLOAT,
    PS/G FLOAT,
    PRIMARY KEY (UNIQUE_ID));

CREATE TABLE NBA_TEAMS
   (teamId VARCHAR(40),
    teamFullName VARCHAR(40),
    teamURLcode VARCHAR(40),
    city VARCHAR(40),
    conferenceName VARCHAR(40),
    divisionName VARCHAR(40),
    PRIMARY KEY (teamId));   

CREATE TABLE NBA_SCHEDULE_GAME_DATE
   (gameId VARCHAR(40),
    startDateEastern VARCHAR(40),
    PRIMARY KEY (gameId));
